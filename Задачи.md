+ Строка В Дату
 
```
Функция СтрокаВДату (Строка)
КвалификаторДаты = Новый КвалификаторыДаты(ЧастиДаты.Дата);
ОписаниеТипДата = Новый ОписаниеТипов("Дата",,,КвалификаторДаты);

МассивСтрок = СтрРазделить(Строка," ",Ложь);

Если МассивСтрок.Количество() >= 3 Тогда 
	
	ДеньЦифрами = МассивСтрок[0];
	Если СтрДлина(ДеньЦифрами) = 1 Тогда
		ДеньЦифрами = "0" + ДеньЦифрами;
	КонецЕсли;
	
	МесяцБуквами = МассивСтрок[1];
		Если ВРег(МесяцБуквами) = "ЯНВАРЯ" Тогда
			МесяцЦифрами = "01";
		ИначеЕсли ВРег(МесяцБуквами) = "ФЕВРАЛЯ" Тогда
			МесяцЦифрами = "02";
		ИначеЕсли ВРег(МесяцБуквами) = "МАРТА" Тогда
			МесяцЦифрами = "03";
		ИначеЕсли ВРег(МесяцБуквами) = "АПРЕЛЯ" Тогда
			МесяцЦифрами = "04";
		ИначеЕсли ВРег(МесяцБуквами) = "МАЯ" Тогда
			МесяцЦифрами = "05";
		ИначеЕсли ВРег(МесяцБуквами) = "ИЮНЯ" Тогда
			МесяцЦифрами = "06";
		ИначеЕсли ВРег(МесяцБуквами) = "ИЮЛЯ" Тогда
			МесяцЦифрами = "07";
		ИначеЕсли ВРег(МесяцБуквами) = "АВГУСТА" Тогда
			МесяцЦифрами = "08";
		ИначеЕсли ВРег(МесяцБуквами) = "СЕНТЯБРЯ" Тогда
			МесяцЦифрами = "09";
		ИначеЕсли ВРег(МесяцБуквами) = "ОКТЯБРЯ" Тогда
			МесяцЦифрами = "10";
		ИначеЕсли ВРег(МесяцБуквами) = "НОЯБРЯ" Тогда
			МесяцЦифрами = "11";
		ИначеЕсли ВРег(МесяцБуквами) = "ДЕКАБРЯ" Тогда
			МесяцЦифрами = "12";
		Иначе
			МесяцЦифрами = Неопределено;
		КонецЕсли;
		
	
	ГодЦифрами = МассивСтрок[2];
	СтрокаСДатой = ГодЦифрами + МесяцЦифрами + ДеньЦифрами;	
	Результат = ОписаниеТипДата.ПривестиЗначение(СтрокаСДатой);
	
	Возврат Результат;
	
	Иначе
	Результат = Неопределено;
КонецЕсли; 

КонецФункции

СтрокаСДатой  = "9 августа 1999 г.";
Дата = СтрокаВДату(СтрокаСДатой);
Сообщить(Дата);
```

```
Функция РабочихИВыходныхДней (ДатаНачала, ДатаОкончания)

	РабочихДней = 0;
	ВыходныхДней = 0;
	
	ТекДата = ДатаНачала;
	Пока ТекДата <= ДатаОкончания Цикл 
		
		НомерДня = ДеньНедели(ТекДата);
		Если НомерДня <= 5 Тогда 
			РабочихДней = РабочихДней + 1;
		Иначе
			ВыходныхДней = ВыходныхДней + 1;
		КонецЕсли;
		
		ТекДата = ТекДата + 86400;
	КонецЦикла;
	   Результат = Новый Структура;
	   Результат.Вставить("РабочихДней", РабочихДней);
	   Результат.Вставить("ВыходныхДней", ВыходныхДней);
	   
	   Возврат Результат

ДатаНачала = НачалоМесяца(ТекущаяДата());
```

```
	МассивТоваров.Добавить(СтруктураТовара);
```
