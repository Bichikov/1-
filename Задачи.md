МассивСтрок = СтрРазделить(Строка," ",Ложь);

	ДеньЦифрами = МассивСтрок[0];
	Если СтрДлина(ДеньЦифрами) = 1 Тогда
		ДеньЦифрами = "0" + ДеньЦифрами;

			МесяцЦифрами = "02";
		ИначеЕсли ВРег(МесяцБуквами) = "МАРТА" Тогда

			МесяцЦифрами = "04";
		ИначеЕсли ВРег(МесяцБуквами) = "МАЯ" Тогда
			МесяцЦифрами = "05";

			МесяцЦифрами = "06";
		ИначеЕсли ВРег(МесяцБуквами) = "ИЮЛЯ" Тогда
			МесяцЦифрами = "07";
		ИначеЕсли ВРег(МесяцБуквами) = "АВГУСТА" Тогда
			МесяцЦифрами = "08";

		ИначеЕсли ВРег(МесяцБуквами) = "НОЯБРЯ" Тогда
			МесяцЦифрами = "11";

			МесяцЦифрами = "12";
		Иначе
			МесяцЦифрами = Неопределено;
		КонецЕсли;

	СтрокаСДатой = ГодЦифрами + МесяцЦифрами + ДеньЦифрами;	
	Результат = ОписаниеТипДата.ПривестиЗначение(СтрокаСДатой);
	
	Возврат Результат;
	
	Иначе
	Результат = Неопределено;
КонецЕсли;

СтрокаСДатой  = "9 августа 1999 г.";
